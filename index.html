Store - localStorage com namespace versionado

const Store = {
  _k: k => PFX + k,
  get(k, fb) {
    try {
      const v = localStorage.getItem(this._k(k));
      return v === null ? fb : JSON.parse(v);
    } catch { return fb; }
  },
  set(k, v) {
    try { localStorage.setItem(this._k(k), JSON.stringify(v)); }
    catch(e) { console.warn('[Store.set]', e); }
  },
  del(k) {
    try { localStorage.removeItem(this._k(k)); }
    catch { }
  },
  dump(state) {
    return {
      app: APP, 
      version: VER,
      exportedAt: new Date().toISOString(),
      data: {
        settings: state.settings,
        breaks: state.breaks,
        activeBreak: state.activeBreak,
        vault: state.vault,
        tasks: state.tasks,
        wikiSearches: state.wikiSearches
      }
    };
  }
};

Estado Reativo

const clone = o => JSON.parse(JSON.stringify(o));
const State = clone(DEFAULT);

function loadState() {
  Object.keys(DEFAULT).forEach(k => {
    State[k] = Store.get(k, clone(DEFAULT[k]));
  });
}

function persist(key) {
  if (key) { Store.set(key, State[key]); return; }
  Object.keys(DEFAULT).forEach(k => Store.set(k, State[k]));
}

const toMins= hhmm => {

const toHHMM= m => {

function applyTheme(t) {
  const theme = t || State.settings.theme || 'dark';
  document.body.className = '';
  if (theme !== 'dark') document.body.classList.add(`theme-${theme}`);
  const meta = document.querySelector('meta[name="theme-color"]');
  if (meta) meta.content = theme === 'light' ? '#F2F2F7' : '#060608';
  $$('#themeGrid .theme-swatch').forEach(s =>
    s.classList.toggle('active', s.dataset.theme === theme)
  );
  $('statTheme').textContent = theme.charAt(0).toUpperCase()+theme.slice(1);
}

function renderAvatar() {
  const av = State.settings.avatar || '⚡';
  const ring = $('profileRing');
  if (ring) {
    ring.textContent = '';
    if (av.startsWith('data:image/')) {
      const img = document.createElement('img');
      img.alt = 'Avatar'; 
      img.src = av;
      ring.appendChild(img);
    } else {
      ring.textContent = av;
    }
  }

  const bioIcon = $('bioIcon');
  if (bioIcon) {
    bioIcon.textContent = '';
    if (av.startsWith('data:image/')) {
      const img = document.createElement('img');
      img.alt = 'av'; 
      img.src = av;
      bioIcon.appendChild(img);
    } else {
      bioIcon.textContent = av;
    }
  }
}

Lock Screen

const Lock = {
  buf: '',
  attempts: 0,
  maxAttempts: 5,
  cooldown: false,

  init() {
    this._buildPad();
    this._renderDots();
  },

  _buildPad() {
    const pad  = $('pinPad');
    const dots = $('pinDots');
    pad.textContent = ''; 
    dots.textContent = '';

    const keys = [
      ['1',''],['2','ABC'],['3','DEF'],
      ['4','GHI'],['5','JKL'],['6','MNO'],
      ['7','PQRS'],['8','TUV'],['9','WXYZ'],
      ['','empty'],['0',''],['⌫','del']
    ];

    keys.forEach(([num, sub]) => {
      const btn = document.createElement('button');
      btn.type = 'button';
      if (sub === 'empty') {
        btn.className = 'pin-key empty'; 
        btn.tabIndex = -1; 
        pad.appendChild(btn); 
        return;
      }
      if (sub === 'del') {
        btn.className = 'pin-key del'; 
        btn.textContent = num;
        btn.addEventListener('click', () => this.back());
      } else {
        btn.className = 'pin-key';
        btn.innerHTML = `${num}${sub ? `${sub}` : ''}`;
        btn.addEventListener('click', () => this.push(num));
      }
      pad.appendChild(btn);
    });
  },

  _renderDots() {
    const pin = String(State.settings.pin || '1234');
    $$('#pinDots .pin-dot').forEach((d, i) =>
      d.classList.toggle('filled', i < Math.min(this.buf.length, pin.length))
    );
  },

  push(n) {
    if (this.cooldown) return;
    const pin = String(State.settings.pin || '1234');
    if (this.buf.length >= pin.length) return;
    this.buf += n;
    this._renderDots();
    if (this.buf.length >= pin.length) setTimeout(() => this._tryUnlock(), 80);
  },

  back() {
    this.buf = this.buf.slice(0, -1);
    this._renderDots();
  },

  _tryUnlock() {
    const pin = String(State.settings.pin || '1234');
    if (this.buf === pin) {
      this.unlock(); 
      return;
    }
    this.attempts++;
    $('lockError').textContent = `PIN incorreto. Tentativa ${this.attempts}/${this.maxAttempts}`;
    navigator.vibrate?.([60,30,60]);
    this.buf = ''; 
    this._renderDots();

    if (this.attempts >= this.maxAttempts) {
      this.cooldown = true;
      $$('.pin-key').forEach(k => k.disabled = true);
      $('lockError').textContent = 'Muitas tentativas. Aguarde 15s.';
      setTimeout(() => {
        this.cooldown = false; 
        this.attempts = 0;
        $$('.pin-key').forEach(k => k.disabled = false);
        $('lockError').textContent = '';
      }, 15000);
    }
  },

  async tryBiometric() {
    if (!State.settings.bioEnabled) return;
    const ring = $('bioRing');
    const icon = $('bioIcon');
    ring?.classList.add('pulse');

    try {
      if (window.PublicKeyCredential && Store.get('bioCredId', '')) {
        const credId = Uint8Array.from(atob(Store.get('bioCredId','')), c => c.charCodeAt(0));
        const assertion = await navigator.credentials.get({
          publicKey: {
            challenge: Uint8Array.from('test', c => c.charCodeAt(0)),
            allowCredentials: [{id: credId, type: 'public-key'}],
            userVerification: 'preferred'
          }
        });
        if (assertion) this.unlock();
      }
    } catch(e) {
      console.log('[Biometric]', e);
    }
  },

  unlock() {
    this.buf = '';
    this._renderDots();
    this.attempts = 0;
    document.body.classList.remove('locked');
    toast('Desbloqueado!');
  }
};
